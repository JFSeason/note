如何查询返回指定的字段。其中exists是判断是否存在该字段，include表示包含该字段，exclude表示不包含该字段

Query query = new Query();
Criteria c = new Criteria();
c.andOperator(Criteria.where("LONG").exists(true), Criteria.where("LAT").exists(true));
query.addCriteria(c);
query.fields().include("LONG").include("LAT").include("_id");
List<TerminalRealData> realData_collection = mongoTemplate.find(query, TerminalRealData.class,"TerminalRealData");
